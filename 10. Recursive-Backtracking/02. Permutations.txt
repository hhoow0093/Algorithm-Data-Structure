class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        res = []
        permute = []
        def create_permuteset():
            if len(permute) == len(nums):
                res.append(permute[:])
            for num in nums:
                if num not in permute:
                    permute.append(num)
                    create_permuteset()
                    permute.pop()
        create_permuteset()
        return res
